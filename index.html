// Google Apps Script for Molecular Lab Reservation System const SHEET_ID = '1yfGdHupYs17o59m-Tmexlu_PDH8OFNsQQnkfCnSDy8s'; function doGet() { return handleRequest(); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { try { let data = {}; if (e && e.postData) { data = JSON.parse(e.postData.contents); } else if (e && e.parameter) { data = e.parameter; } const action = data.action || (e && e.parameter ? e.parameter.action : 'getReservations'); if (action === 'addReservation') { return addReservation(data.reservation); } else if (action === 'getReservations') { return getReservations(); } else if (action === 'addPost') { return addPost(data.post); } else if (action === 'getPosts') { return getPosts(); } else if (action === 'cancelReservation') { return cancelReservation(data.reservationId); } return createResponse({ status: 'error', message: 'Invalid action' }); } catch (error) { return createResponse({ status: 'error', message: error.toString() }); } } function addReservation(reservation) { const ss = SpreadsheetApp.openById(SHEET_ID); let sheet = ss.getSheetByName('Reservations'); if (!sheet) { sheet = ss.insertSheet('Reservations'); sheet.appendRow([ 'ID', 'User Email', 'Room', 'Type', 'Date', 'Start Time', 'End Time', 'Duration (hours)', 'Start DateTime', 'End DateTime', 'Samples', 'Chilex (μL)', 'Proteinase K (μL)', 'TE Buffer (μL)', 'Thermal Cycler', 'Polymerase Type', 'KOD ONE Mix', 'KOD FX', 'dNTP', '2X Buffer', 'Primer', 'ddH₂O (μL)', 'Gel Purpose', 'Gel Notes', 'Is Overnight', 'Is Night', 'Timestamp', 'Status' ]); } sheet.appendRow([ reservation.id, reservation.user, reservation.room, reservation.type, reservation.date, reservation.startTime, reservation.endTime || '', reservation.duration, reservation.startDateTime, reservation.endDateTime, reservation.samples || '', reservation.chilex || '', reservation.proteinaseK || '', reservation.teBuffer || '', reservation.thermalCycler || '', reservation.polymeraseType || '', reservation.kodOneMix || '', reservation.kodFx || '', reservation.dntp || '', reservation.buffer2x || '', reservation.primer || '', reservation.ddh2o || '', reservation.purpose || '', reservation.notes || '', reservation.isOvernight || false, reservation.isNight || false, reservation.timestamp, 'Active' ]); return createResponse({ status: 'success', message: 'Reservation added', id: reservation.id }); } function getReservations() { const ss = SpreadsheetApp.openById(SHEET_ID); const sheet = ss.getSheetByName('Reservations'); if (!sheet) { return createResponse({ status: 'success', reservations: [] }); } const data = sheet.getDataRange().getValues(); const headers = data[0]; const reservations = []; for (let i = 1; i < data.length; i++) { const row = data[i]; if (row[headers.indexOf('Status')] === 'Active') { const reservation = {}; headers.forEach((header, index) => { reservation[header] = row[index]; }); reservations.push(reservation); } } return createResponse({ status: 'success', reservations: reservations }); } function cancelReservation(reservationId) { const ss = SpreadsheetApp.openById(SHEET_ID); const sheet = ss.getSheetByName('Reservations'); if (!sheet) { return createResponse({ status: 'error', message: 'Sheet not found' }); } const data = sheet.getDataRange().getValues(); const headers = data[0]; const idColumn = headers.indexOf('ID'); const statusColumn = headers.indexOf('Status'); for (let i = 1; i < data.length; i++) { if (data[i][idColumn] == reservationId) { sheet.getRange(i + 1, statusColumn + 1).setValue('Cancelled'); break; } } return createResponse({ status: 'success', message: 'Reservation cancelled' }); } function addPost(post) { const ss = SpreadsheetApp.openById(SHEET_ID); let sheet = ss.getSheetByName('Discussion'); if (!sheet) { sheet = ss.insertSheet('Discussion'); sheet.appendRow(['ID', 'User Email', 'User Name', 'Content', 'Image URL', 'Timestamp', 'Comments']); } sheet.appendRow([ post.id, post.user, post.userName, post.content, post.image || '', post.timestamp, JSON.stringify(post.comments || []) ]); return createResponse({ status: 'success', message: 'Post added' }); } function getPosts() { const ss = SpreadsheetApp.openById(SHEET_ID); const sheet = ss.getSheetByName('Discussion'); if (!sheet) { return createResponse({ status: 'success', posts: [] }); } const data = sheet.getDataRange().getValues(); const headers = data[0]; const posts = []; for (let i = 1; i < data.length; i++) { const row = data[i]; const post = {}; headers.forEach((header, index) => { if (header === 'Comments') { post[header] = JSON.parse(row[index] || '[]'); } else { post[header] = row[index]; } }); posts.push(post); } return createResponse({ status: 'success', posts: posts }); } function createResponse(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
